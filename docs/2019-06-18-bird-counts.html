p_entropy_species <- birds %>%
  group_by(year) %>%
  summarize(entropy = entropy(how_many_counted)) %>%
  ggplot(aes(x = year, y = entropy)) + geom_line() +
  labs(x = "Year", y = "Shannon diversity",
       title = "Species diversity per year")
